    <script>
        const apiKey = "";
        const stage = document.getElementById('canvas-stage');
        const editTools = document.getElementById('edit-tools');
        let selectedElement = null;
        let isDragging = false;
        let isResizing = false;
        let startX, startY, startW, startH, startLeft, startTop;

        // --- 檔案上傳 ---
        document.getElementById('imageUpload').addEventListener('change', async (e) => {
            const files = Array.from(e.target.files);
            for (const file of files) {
                const url = await fileToDataUrl(file);
                createLayer(url);
            }
        });

        function fileToDataUrl(file) {
            return new Promise(resolve => {
                const reader = new FileReader();
                reader.onload = e => resolve(e.target.result);
                reader.readAsDataURL(file);
            });
        }

        // --- 建立圖層 ---
        function createLayer(src) {
            const container = document.createElement('div');
            container.className = 'layer-obj';
            container.style.left = '50px';
            container.style.top = '50px';
            container.style.zIndex = stage.querySelectorAll('.layer-obj').length + 1;
            
            // 存儲該圖層的獨立調整參數
            container.dataset.brightness = 100;
            container.dataset.contrast = 100;
            container.dataset.blur = 0;
            container.dataset.filter = 'none';

            const img = document.createElement('img');
            img.src = src;
            img.style.width = '200px'; 
            img.style.display = 'block';
            img.draggable = false;

            const resizer = document.createElement('div');
            resizer.className = 'resizer';

            container.appendChild(img);
            container.appendChild(resizer);
            stage.appendChild(container);

            container.addEventListener('mousedown', (e) => selectElement(e, container));
        }

        // --- 選擇與交互 ---
        function selectElement(e, el) {
            if (selectedElement) selectedElement.classList.remove('active');
            selectedElement = el;
            selectedElement.classList.add('active');
            editTools.classList.remove('hidden');

            // 同步滑桿狀態
            document.getElementById('range-brightness').value = el.dataset.brightness;
            document.getElementById('range-contrast').value = el.dataset.contrast;
            document.getElementById('range-blur').value = el.dataset.blur;
            syncLabels();

            startX = e.clientX;
            startY = e.clientY;
            startLeft = parseInt(el.style.left);
            startTop = parseInt(el.style.top);
            startW = el.offsetWidth;
            startH = el.offsetHeight;

            if (e.target.classList.contains('resizer')) {
                isResizing = true;
            } else {
                isDragging = true;
            }
            
            e.stopPropagation();
        }

        window.addEventListener('mousemove', (e) => {
            if (!selectedElement) return;

            const dx = e.clientX - startX;
            const dy = e.clientY - startY;

            if (isDragging) {
                selectedElement.style.left = `${startLeft + dx}px`;
                selectedElement.style.top = `${startTop + dy}px`;
            } else if (isResizing) {
                const newWidth = Math.max(30, startW + dx);
                selectedElement.querySelector('img').style.width = `${newWidth}px`;
            }
        });

        window.addEventListener('mouseup', () => {
            isDragging = false;
            isResizing = false;
        });

        stage.addEventListener('mousedown', (e) => {
            if (e.target === stage) {
                if (selectedElement) selectedElement.classList.remove('active');
                selectedElement = null;
                editTools.classList.add('hidden');
            }
        });

        // --- 圖片調整邏輯 ---
        function applyCssFilter(filterStr) {
            if (!selectedElement) return;
            selectedElement.dataset.filter = filterStr;
            updateAdjustments();
        }

        function updateAdjustments() {
            if (!selectedElement) return;
            const b = document.getElementById('range-brightness').value;
            const c = document.getElementById('range-contrast').value;
            const blur = document.getElementById('range-blur').value;
            
            selectedElement.dataset.brightness = b;
            selectedElement.dataset.contrast = c;
            selectedElement.dataset.blur = blur;

            const img = selectedElement.querySelector('img');
            const baseFilter = selectedElement.dataset.filter === 'none' ? '' : selectedElement.dataset.filter;
            img.style.filter = `${baseFilter} brightness(${b}%) contrast(${c}%) blur(${blur}px)`;
            syncLabels();
        }

        function syncLabels() {
            document.getElementById('val-brightness').innerText = document.getElementById('range-brightness').value + '%';
            document.getElementById('val-contrast').innerText = document.getElementById('range-contrast').value + '%';
            document.getElementById('val-blur').innerText = document.getElementById('range-blur').value + 'px';
        }

        // --- 功能操作 ---
        function changeZIndex(val) {
            if (!selectedElement) return;
            const currentZ = parseInt(selectedElement.style.zIndex) || 1;
            selectedElement.style.zIndex = Math.max(1, currentZ + val);
        }

        function deleteLayer() {
            if (!selectedElement) return;
            selectedElement.remove();
            selectedElement = null;
            editTools.classList.add('hidden');
        }

        // --- AI 背景移除 ---
        async function aiRemoveBackground() {
            if (!selectedElement) return;
            const img = selectedElement.querySelector('img');
            const btn = document.getElementById('bg-remove-btn');
            const loader = document.getElementById('bg-loader');
            const btnText = document.getElementById('bg-btn-text');

            btn.disabled = true;
            loader.classList.remove('hidden');
            btnText.innerText = "處理中...";

            try {
                // 為了獲取目前的 Canvas 內容(包含濾鏡可能無法直接轉，我們抓原始 src)
                // 實際上去背應該針對原始圖
                const responseUrl = img.src;
                let base64Data;
                if (responseUrl.startsWith('data:')) {
                    base64Data = responseUrl.split(',')[1];
                } else {
                    // 若是外部 URL 需轉 base64，此處假設為上傳的 dataUrl
                    base64Data = responseUrl; 
                }
                
                const prompt = "Remove the background and return only the main subject. Modality: IMAGE.";
                
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-image-preview:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{
                            parts: [
                                { text: prompt },
                                { inlineData: { mimeType: "image/png", data: base64Data } }
                            ]
                        }],
                        generationConfig: { responseModalities: ['TEXT', 'IMAGE'] }
                    })
                });

                const result = await response.json();
                const newBase64 = result.candidates?.[0]?.content?.parts?.find(p => p.inlineData)?.inlineData?.data;

                if (newBase64) {
                    img.src = `data:image/png;base64,${newBase64}`;
                } else {
                    alert("AI 無法辨識主體，請換一張圖片試試。");
                }
            } catch (err) {
                console.error(err);
                alert("背景移除失敗");
            } finally {
                btn.disabled = false;
                loader.classList.add('hidden');
                btnText.innerText = "✨ AI 智慧去背";
            }
        }

        // --- 匯出功能 ---
        async function exportCanvas() {
            const exportCanvas = document.createElement('canvas');
            exportCanvas.width = 800;
            exportCanvas.height = 600;
            const ectx = exportCanvas.getContext('2d');

            ectx.fillStyle = "white";
            ectx.fillRect(0, 0, 800, 600);

            const layers = Array.from(stage.querySelectorAll('.layer-obj'))
                .sort((a, b) => (parseInt(a.style.zIndex)||0) - (parseInt(b.style.zIndex)||0));

            for (const layer of layers) {
                const img = layer.querySelector('img');
                const x = parseInt(layer.style.left);
                const y = parseInt(layer.style.top);
                const w = layer.offsetWidth;
                const h = layer.offsetHeight;
                
                if (!img.complete) await new Promise(r => img.onload = r);
                
                // 匯出時套用濾鏡
                ectx.save();
                ectx.filter = getComputedStyle(img).filter;
                ectx.drawImage(img, x, y, w, h);
                ectx.restore();
            }

            const link = document.createElement('a');
            link.download = `canvas-export-${Date.now()}.png`;
            link.href = exportCanvas.toDataURL('image/png');
            link.click();
        }
    </script>
